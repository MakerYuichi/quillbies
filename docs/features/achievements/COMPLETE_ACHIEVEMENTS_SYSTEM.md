## Helper Functions Available

```typescript
// Get all achievements
getAllAchievements(): Achievement[]

// Get visible challenges
getVisibleChallenges(): Achievement[]

// Get hidden secrets
getHiddenSecrets(): Achievement[]

// Get by challenge type
getDailyChallenges(): Achievement[]
getWeeklyChallenges(): Achievement[]
getMonthlyChallenges(): Achievement[]

// Get by category
getAchievementsByCategory(category: string): Achievement[]

// Get by rarity
getAchievementsByRarity(rarity: string): Achievement[]

// Get unlocked/locked
getUnlockedAchievements(achievements): Achievement[]
getLockedAchievements(achievements): Achievement[]

// Calculate totals
calculateTotalGems(achievements): number
calculateTotalQbies(achievements): number

// Check achievement type
isDailyChallenge(achievementId: string): boolean
isWeeklyChallenge(achievementId: string): boolean
isMonthlyChallenge(achievementId: string): boolean
isHiddenSecret(achievementId: string): boolean
```
